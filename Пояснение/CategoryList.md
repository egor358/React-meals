```jsx
import React from "react";                                            // React + JSX — обязательно для всех компонентов

import CategoryItem from "./CategoryItem";                           // Одна карточка категории (Beef, Chicken и т.д.)

function CategoryList({ catalog }) {                                   // Принимаем пропс catalog — массив из 14 категорий, который пришёл из Home.jsx
  if (!catalog.length) {                                               // Если данные ещё не загрузились или массив пустой
    return <h1>Catalog is missing</h1>;                                // Показываем заглушку, чтобы не было пустой страницы
  }

  return (
    <div className="list">                                             // Контейнер-сетка (в CSS сделана flex/grid, чтобы карточки шли в ряд)
      {catalog.map((cat) => (                                          // Перебираем каждую категорию из массива
        <CategoryItem key={cat.idCategory} {...cat} />                 // Создаём одну карточку
                                                                       // key — обязателен для React (чтобы понимать, что изменилось)
                                                                       // {...cat} — передаём ВСЕ поля категории одной строчкой
      ))}
    </div>
  );
}

export default CategoryList;                                           // Экспортируем, чтобы использовать в Home.jsx
`
Home.jsx получил данные → catalog уже не пустой
CategoryList проверяет длину → всё ок → идёт дальше
.map() создаёт 14 компонентов CategoryItem
Каждый CategoryItem получает свои данные (название, фото, описание)
Пользователь видит ровную сетку карточек
При клике на любую карточку — переход на страницу категории

Почему всё сделано именно так

Один компонент отвечает только за список — чистая архитектура
key и spread {...cat} — стандартный и самый короткий способ передачи данных
Заглушка при пустом массиве — защита от «белого экрана»
