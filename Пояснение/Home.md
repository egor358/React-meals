```jsx
import React from 'react';                                            // Без React не будет JSX
import { useState } from 'react';                                        // Нужен для создания состояния
import { useEffect } from 'react';                                       // Нужен для запроса данных при загрузке
import { getAllCategories } from '../api';                               // Функция → делает запрос на https://www.themealdb.com/api/json/v1/1/categories.php
import Preloader from '../components/Preloader';                         // Красивый спиннер, пока ждём ответ от сервера
import CategoryList from '../components/CategoryList';                   // Компонент, который превращает массив категорий в карточки

function Home() {
  // 1. Состояние для хранения списка категорий (Beef, Chicken, Dessert и т.д.)
  const [catalog, setCatalog] = useState([]);          // изначально — пустой массив

  // 2. Состояние-флаг: идёт ли сейчас загрузка данных
  const [loading, setLoading] = useState(true);        // изначально true → показываем спиннер

  // 3. Этот эффект сработает **один раз** сразу после того, как страница впервые отрисуется
  useEffect(() => {
    (async () => {                                     // Самовызывающаяся асинхронная функция
      try {
        setLoading(true);                              // На всякий случай включаем спиннер ещё раз

        // Делаем запрос к API и ждём готовый объект
        const data = await getAllCategories();         
        // data выглядит так: { categories: [ {idCategory: "1", strCategory: "Beef", ...}, ... ] }

        // Сохраняем в состояние только нужный нам массив
        setCatalog(data.categories);                   
      } catch (err) {
        // Если нет интернета или сервер упал — просто покажем ошибку в консоль
        console.log('Ошибка загрузки категорий:', err);
      } finally {
        // Этот блок выполнится ВСЕГДА — и при успехе, и при ошибке
        setLoading(false);                             // Выключаем спиннер
      }
    })();                                              // ← сразу вызываем функцию
  }, []);                                              // ← пустой массив = выполнить только один раз

  // 4. Что именно показываем пользователю
  return (
    <>
      {/* Условный рендеринг: */}
      {loading 
        ? <Preloader />                                          // Пока грузимся → крутится спиннер
        : <CategoryList catalog={catalog} />                     // Как только всё готово → 14 карточек
      }
    </>
  );
}

export default Home;                                   // Делаем компонент доступным для импорта
`
Зачем нужен этот файл
Это самая первая страница сайта. Именно её видит человек, когда открывает твой проект.
Что происходит по шагам (реальная жизнь)
Пользователь заходит → React монтирует компонент Home
Сразу запускается useEffect (потому что массив зависимостей пустой)
Мы видим спиннер (loading === true)
Уходит запрос на сервер → через 0.3–0.8 сек приходит ответ
setCatalog([...]) → сохраняем 14 категорий
setLoading(false) → спиннер исчезает
React перерисовывает → вместо спиннера появляются красивые карточки категорий
Пользователь кликает, например, на «Beef» → переходит на страницу категории

Почему всё сделано именно так
Один запрос при загрузке → не грузим лишний раз
Спиннер → пользователь понимает, что сайт работает
Чистый и понятный код → любой разработчик разберётся за 10 секунд
